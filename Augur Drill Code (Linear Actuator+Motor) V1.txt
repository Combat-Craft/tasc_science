#define dirPin 11
#define stepPin 10

// IBT-2 pins for linear actuator
const int RPWM = 5;
const int LPWM = 6;

// Speeds
const int ACTUATOR_SPEED = 230;   // 0-255
const unsigned long STEP_INTERVAL = 2000; // microseconds, bigger = slower

char actuatorCommand = 's'; // e = extend, r = retract, s = stop
char stepperCommand = 's';  // l = left, f = right, s = stop

unsigned long lastStepTime = 0;
bool stepState = LOW;

void setup() {
  Serial.begin(9600);

  // Linear actuator setup
  pinMode(RPWM, OUTPUT);
  pinMode(LPWM, OUTPUT);
  analogWrite(RPWM, 0);
  analogWrite(LPWM, 0);

  // Stepper setup
  pinMode(stepPin, OUTPUT);
  pinMode(dirPin, OUTPUT);
  digitalWrite(stepPin, LOW);

  pinMode(LED_BUILTIN, OUTPUT);

  Serial.println("Arduino ready.");
  Serial.println("--- Linear Actuator ---");
  Serial.println("e = extend");
  Serial.println("r = retract");
  Serial.println("x = stop actuator only");
  Serial.println("--- Stepper Motor ---");
  Serial.println("l = stepper left");
  Serial.println("f = stepper right");
  Serial.println("p = stop stepper only");
  Serial.println("s = stop everything");
  Serial.println("");
  Serial.println("Example: press r, then press l = retract AND rotate left");
}

void loop() {
  readCommand();
  runActuator();
  runStepper();
  blinkAlive();
}

void readCommand() {
  if (Serial.available() > 0) {
    char cmd = Serial.read();

    if (cmd == '\n' || cmd == '\r') {
      return;
    }

    Serial.print("Command: ");
    Serial.println(cmd);

    // ----- Linear actuator -----
    if (cmd == 'e' || cmd == 'E') {
      actuatorCommand = 'e';
      Serial.println("> Actuator EXTENDING");
    }

    else if (cmd == 'r' || cmd == 'R') {
      actuatorCommand = 'r';
      Serial.println("> Actuator RETRACTING");
    }

    else if (cmd == 'x' || cmd == 'X') {
      actuatorCommand = 's';
      Serial.println("> Actuator STOPPED");
    }

    // ----- Stepper motor -----
    else if (cmd == 'l' || cmd == 'L') {
      stepperCommand = 'l';
      digitalWrite(dirPin, LOW);
      Serial.println("> Stepper moving LEFT");
    }

    else if (cmd == 'f' || cmd == 'F') {
      stepperCommand = 'f';
      digitalWrite(dirPin, HIGH);
      Serial.println("> Stepper moving RIGHT");
    }

    else if (cmd == 'p' || cmd == 'P') {
      stepperCommand = 's';
      digitalWrite(stepPin, LOW);
      Serial.println("> Stepper STOPPED");
    }

    // ----- Stop everything -----
    else if (cmd == 's' || cmd == 'S') {
      actuatorCommand = 's';
      stepperCommand = 's';

      analogWrite(RPWM, 0);
      analogWrite(LPWM, 0);
      digitalWrite(stepPin, LOW);

      Serial.println("> EVERYTHING STOPPED");
    }

    else {
      Serial.println("Unknown command. Use e, r, x, l, f, p, s");
    }
  }
}

void runActuator() {
  if (actuatorCommand == 'e') {
    // Extend
    analogWrite(RPWM, 0);
    analogWrite(LPWM, ACTUATOR_SPEED);
  }

  else if (actuatorCommand == 'r') {
    // Retract
    analogWrite(RPWM, ACTUATOR_SPEED);
    analogWrite(LPWM, 0);
  }

  else {
    // Stop actuator
    analogWrite(RPWM, 0);
    analogWrite(LPWM, 0);
  }
}

void runStepper() {
  if (stepperCommand == 'l' || stepperCommand == 'f') {
    unsigned long now = micros();

    if (now - lastStepTime >= STEP_INTERVAL) {
      lastStepTime = now;

      stepState = !stepState;
      digitalWrite(stepPin, stepState);
    }
  }

  else {
    digitalWrite(stepPin, LOW);
    stepState = LOW;
  }
}

void blinkAlive() {
  static unsigned long lastBlink = 0;

  if (millis() - lastBlink >= 1000) {
    lastBlink = millis();
    digitalWrite(LED_BUILTIN, !digitalRead(LED_BUILTIN));
  }
}
